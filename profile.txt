function ps {
  Param($Name)
  Get-Process | Where-Object {
	$_.Name -like "*$Name*"
  }
}

function kill {
	Param($Number)
	taskkill /pid $Number
}

# https://winmerge.org/
function fdif {
	Param($Left, $Right) 
	WinMergeU -e $Left $Right
 }
